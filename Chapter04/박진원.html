F(df1 = k, df2 = l). mean: l/l-2, sd: (2(l2)(k+l-2))/(k((l-2)2)(l-4)), l>5


num1 <- c(5,6,10,40)
num2 <- c(6,10,20,45)

fm <- c()
fsd <- c()
for(i in 1:4) {
  fm <- c(fm, num2[i]/(num2[i]-2))
  fsd <- c(fsd, sqrt(((2*(num2[i]^2)*(num1[i]+num2[i]-2))/(num1[i]*((num2[i]-2)^2)*(num2[i]-4)))/4))
}
